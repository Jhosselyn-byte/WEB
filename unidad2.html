🧮 Ejemplo Práctico

Simulación Monte Carlo para estimar π:

import numpy as np
import matplotlib.pyplot as plt

n = 10000
x = np.random.uniform(-1, 1, n)
y = np.random.uniform(-1, 1, n)
inside_circle = (x**2 + y**2) <= 1
pi_estimate = 4 * np.sum(inside_circle) / n
print(f"Estimación de π: {pi_estimate}")
plt.scatter(x[inside_circle], y[inside_circle], s=1, c='blue')
plt.scatter(x[~inside_circle], y[~inside_circle], s=1, c='red')
plt.show()
